mclDeployDirWin=E:/work/tmp
mclDeployDirLinux=/home/laolang/work/tmp

# 默认环境
mclDefaultEnv=dev

6. build.gradle.kts

import org.gradle.api.tasks.testing.logging.TestExceptionFormat

/**
 * 插件列表
 */
plugins {
    groovy
    application
    jacoco
    id("io.spring.javaformat") version "0.0.47"
    // task-tree 和配置缓存冲突, 当需要查看任务依赖时, 再打开
    // id("com.dorongold.task-tree") version "3.0.0"
}

/**
 * 项目坐标
 */
group = "com.laolang.mcl"
version = "0.0.1"

// 环境相关
val mclDeployDirWin: String by project
val mclDeployDirLinux: String by project
val mclDefaultEnv: String by project

/**
 * 全局配置快照 - 提前声明以便各任务共享
 */
val appName = application.applicationName
val projVersion = version.toString()
val isWindows = org.gradle.internal.os.OperatingSystem.current().isWindows
// 统一环境快照
val rawEnv = project.findProperty("env") as String? ?: mclDefaultEnv // default dev
// 如果命令包含 deploy，强制快照为 prd
val finalEnv = if (gradle.startParameter.taskNames.any { it.contains("deploy") }) "prd" else rawEnv




/**
 * prd 禁用测试
 */
if (finalEnv == "prd") {
    tasks.named("check") { dependsOn.remove(tasks.test) }
}

/**
 * 依赖镜像地址
 */
repositories {
    maven("https://maven.aliyun.com/repository/public/")
    maven("https://maven.aliyun.com/repository/spring/")
    mavenLocal()
    mavenCentral()
}

/**
 * 依赖列表
 */
dependencies {
    implementation(libs.picocli)
    implementation(libs.guice)

    compileOnly(libs.lombok)
    annotationProcessor(libs.lombok)
    implementation(libs.guava)

    implementation(libs.bundles.log4j2)

    testCompileOnly(libs.lombok)
    testAnnotationProcessor(libs.lombok)
    testImplementation(libs.bundles.junit)
    testImplementation(libs.bundles.spock)

    testRuntimeOnly("org.junit.platform:junit-platform-launcher")
}

/**
 * 指定版本
 */
java {
    toolchain {
        languageVersion = JavaLanguageVersion.of(8)
    }
}

/**
 * 指定编码
 */
tasks.withType<JavaCompile>().configureEach {
    options.encoding = "UTF-8"
    // 大项目开启
    // options.isFork = true
}

/**
 * 如果使用了 Groovy (Spock)
 */
tasks.withType<GroovyCompile>().configureEach {
    options.encoding = "UTF-8"
}

/**
 * 项目配置
 */
application {
    mainClass = "com.laolang.mcl.MclApplication"
    applicationName = "mcl"
}

/**
 * 配置资源目录
 */
sourceSets {
    main {
        resources {
            exclude("dev/**", "prd/**")
            srcDirs("src/main/resources/$finalEnv")
        }
    }
}

/**
 * 输出 profile 配置 (兼容配置缓存版)
 */
// 1. 先在外部提取数据，不要在闭包里直接引用 sourceSets
val resourceDirsProvider = provider {
    sourceSets.main.get().resources.srcDirs.filter { it.exists() }.map { it.absolutePath }
}

tasks.named<ProcessResources>("processResources") {
    val dirsToPrint = resourceDirsProvider
    val finalEnv = finalEnv

    doFirst {
        logger.lifecycle("==============================================")
        logger.lifecycle("profile: $finalEnv")
        logger.lifecycle("resources dir: ${dirsToPrint.get()}")
        logger.lifecycle("==============================================")
    }
}

/**
 * 配置测试任务
 */
tasks.named<Test>("test") {
    useJUnitPlatform()

    // 输出详细日志
    testLogging {
        // 记录日志的事件类型
        events("FAILED", "PASSED", "SKIPPED", "STANDARD_ERROR", "STANDARD_OUT", "STARTED")
        // 记录测试异常的格式
        // FULL: 完整显示异常
        // SHORT: 异常的简短显示
        exceptionFormat = TestExceptionFormat.FULL
        // 是否记录标准输出和标准错误的输出
        showStandardStreams = true
    }

    finalizedBy(tasks.jacocoTestReport)
}

/**
 * jacoco 任务
 */
tasks.named<JacocoReport>("jacocoTestReport") {
    // 依赖于测试任务
    dependsOn(tasks.test)

    reports {
        // 把不需要的报告去掉
        xml.required.set(false)
        csv.required.set(false)

        // 只启用 html 报告
        html.required.set(true)

        // jacoco 报告位置
        html.outputLocation.set(layout.buildDirectory.dir("jacocoHtml"))
    }
}

/**
 * 禁用默认的 startScripts 任务, 使用自定义的启动脚本
 */
tasks.named<CreateStartScripts>("startScripts") {
    enabled = false
}

/**
 * 配置分发包任务
 */
distributions {
    main {
        contents {
            // 1. 将你自己的启动脚本放入 bin
            from("res/distributions/bin") {
                into("bin")
                // 赋予 Linux 脚本执行权限 (rwxr-xr-x)
                filesMatching(listOf("*.sh", "mcl")) {
                    permissions {
                        unix(0x1ED) // 0755 的十六进制或直接用字符串
                    }
                }
                filesNotMatching(listOf("*.sh", "mcl")) {
                    permissions {
                        unix(0x1A4) // 0644
                    }
                }
            }

            // 2. 将配置文件放入 conf
            from("src/main/resources/$finalEnv") {
                include("**/log4j2.xml")
                into("conf")
            }

            // 3. 创建空的 logs 目录
            into("logs") {
                from(files())
            }
        }
    }
}

// 解决资源重复问题：从默认的 jar 中排除已经放在 conf 的配置文件
// 这样在运行 jar 时，它会去 classpath (即 conf 目录) 找配置
tasks.named<Jar>("jar") {
    exclude("log4j2.xml")
}


/**
 * 自动化部署任务：替代 maven-antrun-plugin
 * 执行逻辑：编译打包 -> 安装到本地临时目录 -> 同步到最终开发目录
 */
val deploy = tasks.register<Sync>("deploy") {
    group = "distribution"
    description = "解压并同步分发包到指定目录"

    val appName = appName
    val projVersion = projVersion
    val isWindows = isWindows
    val finalEnv = finalEnv
    val mclDeployDirWin = mclDeployDirWin
    val mclDeployDirLinux = mclDeployDirLinux


    val targetName = "${appName}-${projVersion}"
    val targetDir = if (isWindows) "$mclDeployDirWin/$targetName" else "$mclDeployDirLinux/$targetName"

    val installTask = tasks.named<Sync>("installDist")
    dependsOn(installTask)
    from(installTask.map { it.destinationDir })
    into(targetDir)

    preserve {
        include("logs/**") // 同步时不要删掉目标目录里的日志
    }

    doLast {
        logger.lifecycle("==============================================")
        logger.lifecycle("Deploy Action Executed!")
        logger.lifecycle("Environment: $finalEnv")
        logger.lifecycle("Target: $targetDir")
        val checkFile = File("$targetDir/bin/mcl") // 这里其实应该是 $targetDir/bin/$currentAppName
        logger.lifecycle("Binary Exists: ${checkFile.exists()}")
        logger.lifecycle("==============================================")
    }
}

7. 启动脚本

linux
#!/bin/bash

SCRIPT_PATH=$(readlink -f "$0")
SCRIPT_DIR=$(dirname "$SCRIPT_PATH")
BASE_DIR=$(dirname "$SCRIPT_DIR")
WORK_DIR=$(pwd)
LIB_DIR="$BASE_DIR/lib"
CLASSPATH="$BASE_DIR/conf:$LIB_DIR/*"

java \
  -Dmcl.home="$BASE_DIR" \
  -Duser.work.dir="$WORK_DIR" \
  -cp "$CLASSPATH" \
  com.laolang.mcl.MclApplication "$@"
windows
@echo off
chcp 65001 >nul

setlocal

:: 1. Get script directory
set SCRIPT_DIR=%~dp0

:: Remove trailing backslash
if "%SCRIPT_DIR:~-1%"=="\" set SCRIPT_DIR=%SCRIPT_DIR:~0,-1%

:: Calculate parent directory and normalize to absolute path
for %%a in ("%SCRIPT_DIR%\..") do set BASE_DIR=%%~fa

:: 2. Get current working directory
set WORK_DIR=%CD%

:: 3. Setup classpath
set LIB_DIR=%BASE_DIR%\lib
set CLASSPATH=%BASE_DIR%\conf;%LIB_DIR%\*

:: 4. Start Java
java -Dfile.encoding=UTF-8 -Dsun.stdout.encoding=UTF-8 -Dsun.stderr.encoding=UTF-8 ^
-Dmcl.home="%BASE_DIR%" ^
-Duser.work.dir="%WORK_DIR%" ^
-cp "%CLASSPATH%" com.laolang.mcl.MclApplication %*

endlocal
